# HXRRRRRI-UI Package

## Install

```bash
npm install framer-motion
```

## Use

```tsx
import { ThemeProvider, HxButton } from './hxrrrrri-ui'

export function App() {
  return (
    <ThemeProvider>
      <HxButton system="luxury" variant="gradient">Get Started</HxButton>
    </ThemeProvider>
  )
}
```
